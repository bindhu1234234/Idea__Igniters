import tensorflow as tf from tensorflow.keras import layers, models from tensorflow.keras.preprocessing.image import ImageDataGenerator # 1. Setup Hyperparameters BATCH_SIZE = 32 IMG_SIZE = (224, 224) EPOCHS = 10 # Adjust based on time remaining! DATASET_PATH = 'path/to/your/plant_dataset' # 2. Data Augmentation (Crucial for small hackathon datasets) train_datagen = ImageDataGenerator( rescale=1./255, rotation_range=20, zoom_range=0.2, horizontal_flip=True, validation_split=0.2 ) train_generator = train_datagen.flow_from_directory( DATASET_PATH, target_size=IMG_SIZE, batch_size=BATCH_SIZE, class_mode='categorical', subset='training' ) validation_generator = train_datagen.flow_from_directory( DATASET_PATH, target_size=IMG_SIZE, batch_size=BATCH_SIZE, class_mode='categorical', subset='validation' ) # 3. Build the Model (Using MobileNetV2 for speed) base_model = tf.keras.applications.MobileNetV2( input_shape=(224, 224, 3), include_top=False, weights='imagenet' ) base_model.trainable = False # Freeze the pre-trained weights model = models.Sequential([ base_model, layers.GlobalAveragePooling2D(), layers.Dropout(0.2), layers.Dense(train_generator.num_classes, activation='softmax') ]) model.compile(optimizer='adam', loss='categorical_crossentropy', metrics=['accuracy']) # 4. Training print("Starting training...") history = model.fit( train_generator, validation_data=validation_generator, epochs=EPOCHS ) # 5. Save the model for your demo app model.save('plant_disease_model.h5') print("Model saved as plant_disease_model.h5")
