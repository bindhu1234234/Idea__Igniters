import cv2 import numpy as np import tensorflow as tf from tensorflow.keras.models import load_model # Load trained plant disease model disease_model = load_model("plant_disease_model.h5") # Load leaf detector (pretrained MobileNet model) leaf_model = tf.keras.applications.MobileNetV2(weights="imagenet") # Image size IMG_SIZE = 224 # Function to check if image is a leaf def is_leaf(image_path): img = tf.keras.preprocessing.image.load_img(image_path, target_size=(224, 224)) img_array = tf.keras.preprocessing.image.img_to_array(img) img_array = tf.keras.applications.mobilenet_v2.preprocess_input(img_array) img_array = np.expand_dims(img_array, axis=0) predictions = leaf_model.predict(img_array) decoded = tf.keras.applications.mobilenet_v2.decode_predictions(predictions, top=3)[0] # Check if prediction contains plant/leaf for _, label, prob in decoded: if "leaf" in label or "plant" in label or "tree" in label: return True return False # Function to predict disease def predict_disease(image_path): # First check if image is leaf if not is_leaf(image_path): print("❌ This is not a plant leaf image.") return # Load image img = cv2.imread(image_path) img = cv2.resize(img, (IMG_SIZE, IMG_SIZE)) img = img / 255.0 img = np.reshape(img, [1, IMG_SIZE, IMG_SIZE, 3]) # Predict prediction = disease_model.predict(img) class_index = np.argmax(prediction) classes = [ "Tomato Early Blight", "Tomato Late Blight", "Healthy Leaf", "Potato Early Blight", "Potato Late Blight" ] print("🌿 Disease Prediction:", classes[class_index]) # Test predict_disease("test_leaf.jpg")
