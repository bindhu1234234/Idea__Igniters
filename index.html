from flask import Flask, render_template, request import torch from torchvision import transforms from PIL import Image import os app = Flask(__name__) UPLOAD_FOLDER = 'static/uploads' app.config['UPLOAD_FOLDER'] = UPLOAD_FOLDER # Load your trained model model = torch.load('model.pth', map_location='cpu') model.eval() classes = ['Healthy', 'Leaf Spot', 'Rice Blast'] transform = transforms.Compose([ transforms.Resize((224, 224)), transforms.ToTensor() ]) @app.route('/', methods=['GET', 'POST']) def index(): if request.method == 'POST': file = request.files filepath = os.path.join(app.config['UPLOAD_FOLDER'], file.filename) file.save(filepath) # Process image img = Image.open(filepath) img = transform(img).unsqueeze(0) with torch.no_grad(): output = model(img) prob = torch.nn.functional.softmax(output[0], dim=0) pred = classes[prob.argmax().item()] confidence = prob.max().item() return render_template('index.html', prediction=pred, confidence=round(confidence*100, 2), image=filepath) return render_template('index.html') if __name__ == '__main__': app.run(debug=True)
